\begin{ex}%[2H2N2-2]
	Cho tứ diện đều $ ABCD $ có độ dài cạnh bằng $ a $, gọi $ M $ là trung điểm của đoạn thẳng $ CD $. Tích vô hướng $ \vec{AB}\cdot\vec{AM} $ bằng
	\choice
	{$ \dfrac{a^{2}}{4} $}
	{\True $ \dfrac{a^{2}}{2} $}
	{$ \dfrac{a^{2}}{3} $}
	{$ a^{2} $}
	\loigiai{
		\begin{center}
			\begin{tikzpicture}[scale=1, font=\footnotesize, line join=round, line cap=round, >=stealth]
				\def\ac{4} % cạnh AC
				\def\ab{2} % cạnh AB
				\def\h{4} % chiều cao
				\def\gocA{50} % góc A của đáy
				\coordinate[label=left:$B$] (A) at (0,0);
				\coordinate[label=right:$D$] (C) at (\ac,0);
				\coordinate[label=below left:$C$] (B) at (-\gocA:\ab);
				\coordinate[label=below right:$M$] (M) at ($(B)!.5!(C)$);
				\coordinate[label=below:$G$] (G) at ($(A)!2/3!(M)$);
				\coordinate[label=above:$A$] (S) at ($(G)+(90:\h)$);
				\draw (A)--(B)--(C)--(S)--cycle (S)--(B) (S)--(M) ;
				\draw[dashed] (M)--(A)--(C) (G)--(S);
				\foreach \diem in {A,B,C,S,G,M}	\fill (\diem)circle(1.5pt);
			\end{tikzpicture}
		\end{center}
		Vì tứ diện $ ABCD $ là tứ diện đều nên chân đường cao là trọng tâm của tam giác $ BCD $ suy ra $ AG\perp (BCD) \Rightarrow AG\perp CD$.\\
		Tam giác $ BCD $ đều có $ M $ là trung điểm $ CD $ nên $ BM\perp CD $.\\
		Suy ra $ \heva{&CD\perp BM\\&CD\perp AG}\Rightarrow CD\perp (ABM)\Rightarrow CD\perp AB \Rightarrow DM \perp AB \Rightarrow \vec{AB}\cdot \vec{DM} = 0$.\\
		Ta có $ \vec{AB}\cdot\vec{AM} =\vec{AB}\cdot\left(\vec{AD}+\vec{DM}\right)=\vec{AB}\cdot \vec{AD}+\vec{AB}\cdot \vec{DM} =\vec{AB}\cdot \vec{AD}$.\\
		$ \vec{AB}\cdot\vec{AD}=\left|\vec{AB}\right|\cdot\left|\vec{AD}\right|\cdot\cos \left(\vec{AB};\vec{AD}\right)=a\cdot a\cdot \cos 60^{\circ}=\dfrac{a^{2}}{2} $.
	}
\end{ex}

	Cho tứ diện $ABCD$. Khi đó
	\choiceTF
	{$\vec{A B}+\vec{C D}=\vec{A C}+\vec{D B}$}
	{$\vec{A C}+\vec{B D}=\vec{A B}+\vec{C D}$}
	{$\vec{A D}+\vec{B C}=\vec{A B}+\vec{D C}$}

	\loigiai{
\begin{itemchoice}
\itemch Sai. $\vec{A B}+\vec{C D}=\vec{A C}+\vec{D B}\Leftrightarrow\vec{C A}+\vec{A B}+\vec{B D}=\vec{D C}\Leftrightarrow\vec{C D}=\vec{D C}$.
\itemch Sai. $\vec{A C}+\vec{B D}=\vec{A B}+\vec{C D}\Leftrightarrow\vec{A C}-\vec{A B}=\vec{C D}+\vec{D B}\Leftrightarrow\vec{B C}=\vec{C B}$.
\itemch Sai. $\vec{A D}+\vec{B C}=\vec{A B}+\vec{D C}\Leftrightarrow\vec{A D}-\vec{A B}=\vec{D C}+\vec{C B}\Leftrightarrow\vec{B D}=\vec{D B}$.
\itemch Đúng. $\vec{B A}+\vec{C D}=\vec{B D}+\vec{C A}\Leftrightarrow\vec{B A}-\vec{B D}=\vec{D C}+\vec{C A}\Leftrightarrow\vec{D A}=\vec{D A}$.
		\end{itemchoice}
	}
\end{ex}
