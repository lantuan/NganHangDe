\begin{ex}%[2D1H4-1]Câu 7
	Hàm số nào sau đây có một đường tiệm cận?
\choice
{$y=\dfrac{x+3}{2x-1}$}
{$y=\dfrac{x^2+3x-2}{x+3}$}
{\True $y=\dfrac{2x}{x^2+1}$}
{$y=\dfrac{4}{x-1}$}
\loigiai{
Ta xét hàm số $y=\dfrac{x+3}{2x-1}$ có\\
$\bullet \underset{x \to +\infty}{\lim}\,y=\dfrac{1}{2};\underset{x \to -\infty}{\lim}\,y=\dfrac{1}{2} \Rightarrow y=\dfrac{1}{2}$ là đường tiệm cận ngang của đồ thị hàm số.\\
$\bullet \underset{x \to \frac{1}{2}^+}{\lim}\,y=+\infty;\underset{x \to \frac{1}{2}^-}{\lim}\,y=-\infty \Rightarrow x=\dfrac{1}{2}$ là đường tiệm cận đứng của đồ thị hàm số.\\
Ta xét hàm số $y=\dfrac{x^2+3x-2}{x+3}=x-\dfrac{2}{x+3}$ có\\
$\bullet \underset{x \to -3^+}{\lim}\,y=-\infty;\underset{x \to -3^-}{\lim}\,y=+\infty \Rightarrow x=-3$ là đường tiệm cận đứng của đồ thị hàm số.\\
$\bullet a_1=\underset{x \to +\infty}{\lim}\,\dfrac{\dfrac{x^2+3x-2}{x+3}}{x}=1 ;a_2=\underset{x \to -\infty}{\lim}\,\dfrac{\dfrac{x^2+3x-2}{x+3}}{x}=1 \Rightarrow a=1$\\
$\bullet b_1=\underset{x \to +\infty}{\lim}\,\left( \dfrac{x^2+3x-2}{x+3}-x\right)=0 ; b_2=\underset{x \to -\infty}{\lim}\,\left( \dfrac{x^2+3x-2}{x+3}-x\right) =0 \Rightarrow b=0$\\
$\Rightarrow y=ax+b=x$ là đường tiệm cận xiên của đồ thị hàm số.\\
$\bullet$Vì bậc của đa thức tử lớn hơn bậc của đa thức mẫu nên đồ thị hàm số không có đường tiệm cận ngang.\\
Ta xét hàm số $y=\dfrac{2x}{x^2+1}$ có\\
$\bullet \underset{x \to +\infty}{\lim}\,y=0$ và $\underset{x \to -\infty}{\lim}\,y=0 \Rightarrow y=0$ là đường tiệm cận ngang của đồ thị hàm số.\\
Vì phương trình của mẫu thức bằng $0$ vô nghiệm nên đồ thị hàm số không có đường tiệm cận đứng.\\
Ta xét hàm số $y=\dfrac{4}{x-1}$ có\\
$\bullet \underset{x \to +\infty}{\lim}\,y=0$ và $\underset{x \to -\infty}{\lim}\,y=0 \Rightarrow y=0$ là đường tiệm cận ngang của đồ thị hàm số.\\
$\bullet \underset{x \to 1^+}{\lim}\,y=+\infty;\underset{x \to 1^-}{\lim}\,y=-\infty \Rightarrow x=1$ là đường tiệm cận đứng của đồ thị hàm số.}
\end{ex}

\begin{ex}%[2D1H4-3]Câu 10
	Cho hàm số $y=\dfrac{x-1}{x^2-3x+2}$. Xét các mệnh đề sau\\
	(I). Đồ thị hàm số có tiệm cận ngang $y=0$.\\
	(II). Đồ thị hàm số có hai tiệm cận đứng $x=1$ và $x=2$.\\
	(III). Đồ thị hàm số không có tiệm cận xiên.\\
	Mệnh đề nào sau đây là đúng?

{Chỉ có (III) đúng}

{Cả 3 mệnh đề đều đúng}
\loigiai{Ta có $y=\dfrac{x-1}{x^2-3x+2}$. Xét:\\
$\bullet \underset{x \to \pm \infty}{\lim}\,y=0$. Vậy đồ thị hàm số có tiệm cận ngang là $y=0$. Do đó (I) đúng.\\
$\bullet \underset{x \to 2}{\lim}\,y=\infty;\underset{x \to 1}{\lim}\,y=\infty$\\
Vậy đồ thị hàm số chỉ có một tiệm cận đứng là $x=2$. Do đó (II) sai.\\
Vì bậc tử thức bé hơn bậc mẫu thức nên đồ thị hàm số không có tiệm cận xiên. Do đó (III) đúng.}
\end{ex}

\begin{ex}%[2D1H4-3]Câu 1
	Cho hàm số $y=\dfrac{x^2-2x+2}{x+2}$. Các mệnh đề sau đúng hay sai?

{\True Hàm số có hai đường tiệm cận}
{\True Giao điểm của hai tiệm cận là $I(-2;-6)$}

\loigiai{
	\begin{itemchoice}
		\itemch Đúng. Vì $y=\dfrac{x^2-2x+2}{x+2}=x-4+\dfrac{10}{x+2}$, Ta có:\\ $\bullet \underset{x \to -2}{\lim}\,y=\pm\infty$ nên $x=-2$ là tiệm cận đứng của đồ thị hàm số.\\
		$\bullet \underset{x \to \pm\infty}{\lim}\,\dfrac{10}{x+2}=0$ nên $y=x-4$ là đường tiệm cận xiên của đồ thị hàm số.
		\itemch Đúng. Vì giao điểm của hai tiệm cận thỏa mãn $\heva{&x=-2\\&y=x-4}\Leftrightarrow \heva{&x=-2\\&y=-6}$
		\itemch Sai. Gọi $(C) \colon y=x-4$ khoảng cách từ $O$ đến tiệm cận xiên bằng\\ $\mathrm{d}(O;(c))=\dfrac{\left| -4 \right|}{\sqrt{(-1)^2+1^2}}=\dfrac{4}{\sqrt2}.$
		\itemch Đúng. Vì $M(0;-4) \in (C) \colon y=x-4 \Rightarrow -4=0-4.$
	\end{itemchoice}
		}
\end{ex}

\begin{ex}%[2D1H4-3]Câu 2
