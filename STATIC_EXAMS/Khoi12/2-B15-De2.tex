			\itemch Đúng. \\ $A'$ đối xứng với $A$ qua $(P)$ nên $AA'$ vuông góc với $(P)$.\\

			&\Rightarrow& 41 t-41=0\\
			&\Rightarrow& t=1 \\
			&\Rightarrow& H(5 ; 1 ; 7).
			\end{eqnarray*}
