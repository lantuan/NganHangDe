	\loigiai{
		\begin{itemchoice}

			Suy ra phương trình $(2)$ có $2$ nghiệm.\\
			Vậy phương trình $2f^2(x)-3f(x)+1=0$ có $3$ nghiệm.	
		\end{itemchoice}
