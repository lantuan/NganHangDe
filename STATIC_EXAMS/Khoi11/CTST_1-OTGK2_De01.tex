\begin{bt}%[1D6H4-3]%[1D6H4-2]%[Dự án đề kiểm tra Toán 11 GHKII NH23-24- Tư Đô Nguyên]%[THPT Nam Kì Khởi Nghĩa - TPHCM]
	Giải các phương trình, bất phương trình sau
	\begin{listEX}[2]
		\item $6^{x^2-2}=36$;
		\item $\log_5 \left(4x-5\right)=3$; 
		\item $\left(\dfrac{1}{3} \right)^x \le 3\cdot 9^{2x}$;
		\item  $\log _{\tfrac{1}{2}} \left(x+1\right)\le \log _{\tfrac{1}{2}} \left(2-4x\right)$.
	\end{listEX}
	\loigiai{
	\begin{listEX}[1]
		\item $6^{x^2-2}=36\Leftrightarrow 6^{x^2-2}=6^2 \Leftrightarrow x^2-2=2 \Leftrightarrow \hoac{&x=2 \\ &x=-2.}$
		\item 
		$\log _5(4 x-5)=3 \Leftrightarrow 4 x-5=5^3 \Leftrightarrow x=\dfrac{65}{2}$.
		\item $\left(\dfrac{1}{3}\right)^x \leq 3\cdot 9^{2 x} 
		\Leftrightarrow\left(3^{-1}\right)^x \leq 3 \cdot \left(3^2\right)^{2 x} 
		\Leftrightarrow 3^{-x} \leq 3^{4 x+1} 
		\Leftrightarrow-x \leq 4 x+1 
		\Leftrightarrow x \geq-\dfrac{1}{5}.$
		\item 
			$\log _{\tfrac{1}{2}}(x+1) \leq \log _{\tfrac{1}{2}}(2-4x) \Leftrightarrow\heva{&x+1\geq 2-4x \\
				&2-4x>0} 
				\Leftrightarrow
		\heva{&x \geq \dfrac{1}{5} \\
		&x<\dfrac{1}{2}}  \Leftrightarrow \dfrac{1}{5} \leq x<\dfrac{1}{2}$.
	\end{listEX}
	}
\end{bt}

Hỏi ở độ sâu $10$ mét thì cường độ ánh sáng giảm đi bao nhiêu lần so với cường độ ánh sáng lúc ánh sáng bắt đầu đi vào môi trường nước biển?
	\loigiai{
	Khi ánh sáng bắt đầu đi vào môi trường nước biển thì $x=0$ nên $I_t=I_0$.\\

	Vậy $I_s=I_t \cdot \mathrm{e}^{-14}=\dfrac{I_t}{\mathrm{e}^{14}}$ nên ở độ sâu $10$ mét thì cường độ ánh sáng giảm đi $\mathrm{e}^{14}$ lần.
	}
\end{bt}
