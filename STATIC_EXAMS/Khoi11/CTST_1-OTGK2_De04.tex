\begin{eqnarray*}
&&\log(10a)+\log(10a)^2+\cdots+\log(10a)^{2024}\\
&=&\log(10a)+2\log(10a)+\cdots+2024\log(10a)\\
&=&(1+2+\cdots+2024)\cdot (1+\log a)\\

\end{eqnarray*}
