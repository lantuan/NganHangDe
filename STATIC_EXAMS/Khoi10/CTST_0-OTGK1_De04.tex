\begin{ex}%[0H4N2-1]
	Cho tam giác $ABC$ có $AB=7$ cm, $AC=3$ cm và $\widehat{A}=60^\circ$. Tính độ dài cạnh $BC$.
	\choice
	{$BC=5{,}58$}
	{$BC=9{,}08$}
	{$BC=7{,}08$}

	\loigiai{
		Áp dụng Định lí cô-sin trong tam giác $ABC$ ta có
		$$BC^2=AB^2+AC^2-2AB \cdot AC \cdot \cos A=49+9-2\cdot 7\cdot 3\cdot \dfrac{1}{2}=37.$$

	}
